29def timer_func(): 

30 """ 

31 Prints progress on inserting elements. 

32 """ 

33 

34 last = time.time() 

35 

36 def elapsed(msg): 

37 nonlocal last 

38 now = time.time() 

39 sec = round(now - last, 1) 

40 last = now 

41 print(f"{msg} in {sec} seconds") 

42 

43 def progress(index, step=500, total=0, msg=""): 

44 nonlocal last 

45 if index % step == 0: 

46 percent = int((index / total) * 100) if total >= index else index 

47 elapsed(f"... {percent}% ({index} out of {total}). {step} {msg}") 

48 

49 return elapsed, progress

219def index_posts(posts, ix=None, overwrite=False, add_func=add_index): 

220 """ 

221 Create or update a search index of posts. 

222 """ 

223 

224 ix = ix or init_index() 

225 # The writer is asynchronous by default 

226 writer = AsyncWriter(ix) 

227 

228 elapsed, progress = timer_func() 

229 total = posts.count() 

230 stream = islice(zip(count(1), posts), None) 

231 

232 # Loop through posts and add to index 

233 for step, post in stream: 

234 progress(step, total=total, msg="posts indexed") 

235 add_func(post=post, writer=writer) 

236

238 if overwrite: 

239 logger.info("Overwriting the old index") 

240 writer.commit(mergetype=writing.CLEAR) 

241 else: 

242 writer.commit() 

243 

244 elapsed(f"Indexed posts={total}")

272def preform_whoosh_search(query, ix=None, fields=None, page=None, per_page=None, sortedby=[], **kwargs): 

273 """ 

274 Query the indexed, looking for a match in the specified fields. 

275 Results a tuple of results and an open searcher object. 

276 """ 

277 

278 per_page = per_page or settings.SEARCH_RESULTS_PER_PAGE 

279 fields = fields or ['tags', 'title', 'author', 'author_uid', 'author_handle'] 

280 ix = ix or init_index() 

281 searcher = ix.searcher() 

282 

283 # Splits the query into words and applies 

284 # and OR filter, eg. 'foo bar' == 'foo OR bar' 

285 orgroup = OrGroup 

286 

287 parser = MultifieldParser(fieldnames=fields, schema=ix.schema, group=orgroup).parse(query) 

288 if page: 

289 # Return a pagenated version of the results. 

290 results = searcher.search_page(parser, 

291 pagenum=page, pagelen=per_page, sortedby=sortedby, 

292 reverse=True, 

293 terms=True, **kwargs) 

294 results.results.fragmenter.maxchars = 100 

295 # Show more context before and after 

296 results.results.fragmenter.surround = 100 

297 else: 

298 results = searcher.search(parser, limit=settings.SEARCH_LIMIT, sortedby=sortedby, reverse=True, 

299 terms=True, **kwargs) 

300 # Allow larger fragments 

301 results.fragmenter.maxchars = 100 

302 results.fragmenter.surround = 100 

303 

304 #logger.info("Preformed index search") 

305 

306 return results
