348 def update_parent_counts(self): 

349 """ 

350 Update the counts for the parent and root 

351 """ 

352 

353 descendants = Post.objects.filter(root=self.root).exclude(Q(pk=self.root.pk) | Q(status=Post.DELETED) 

354 | Q(spam=Post.SPAM)) 

355 answer_count = descendants.filter(type=Post.ANSWER).count() 

356 comment_count = descendants.filter(type=Post.COMMENT).count() 

357 reply_count = descendants.count() 

358 # Update the root reply, answer, and comment counts. 

359 Post.objects.filter(pk=self.root.pk).update(reply_count=reply_count, answer_count=answer_count, 

360 comment_count=comment_count) 

361 

362 children = Post.objects.filter(parent=self.parent).exclude(pk=self.parent.pk) 

363 com_count = children.filter(type=Post.COMMENT).count() 

364 

365 # Update parent reply, answer, and comment counts. 

366 Post.objects.filter(pk=self.parent.pk, is_toplevel=False).update(comment_count=com_count, answer_count=0, 

367 reply_count=children.count())
